async void ExecuteClickCommand()
{
    // Show busy indicator
    IsBusy = true;

    // Check if we browsed a file
    if (string.IsNullOrEmpty(FilePath))
        return;

    // Check if browsed file exist or not
    if (File.Exists(FilePath) == false)
        return;

    await Task.Run(() =>
    {
        // Delete selected file.
        File.Delete(FilePath);
    });

    // Hide busy indicator
    IsBusy = false;
}
